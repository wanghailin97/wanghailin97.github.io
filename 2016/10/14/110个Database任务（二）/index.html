database:mydatabase

id int(10),name varchar(10),

create table emp(
eid int(10),ename varchar(10),
title varchar(10),hiretime varchar(10),
salary double,bonus double,dieader varchar(10))set utf8;

D023.条件查询

要求找出学生id为1或者3或者5的学生

use mydatabase;
select* from student
where id=1 or id=3 or id=5;

D024.条件查询

查出区间落在180,190身高之间的学生

select* from student
where height>=180 and height<=190;

D025.根据性别分组

use mydatabase;
select* from student
group by gender;

D026.查询学生的姓名、身高，默认以身高的升序排序

use mydatabase;
select id height from student order by height;

D027.求出所有班级人数大于2的学生人数和班级id

学生表中的字段有id,name(姓名),number(学号),sex(性别),c_id（班级）,插入几组自定义的数据，查询所有班级人数大于2的学生人数和班级id。

use mydatabase;
select class,count(*) from student
group by class having count(*)>2;

D028.查询学生，限制人数为2

use mydatabase;
select* from student limit 2;

D029.查询学生表和班级表

多表查询my_student表和my_class表，使用where 条件查询my_student表中的c_id = my_class表中id的记录。

use mydatabase;
select* from student,class
where class = class_id;

D030.查询学生表和班级表

多表查询my_student表和my_class表，使用左连接查询my_student表中的c_id = my_class表中id的记录。

use mydatabase;
select* from student
left join class on class=class_id;

D032.查询员工姓名及年薪

use mydatabase;
select ename,salary from emp;

D033.查询月工资大于4000的员工信息

select* from emp
where salary>4000;

D034.查询年薪大于20000的员工信息

select* from emp
where 12*salary>20000;

D035.查询没有奖金的员工信息

select* from emp
where bonus is null;

D036.查询月工资大于3000同时有奖金的员工信息

select* from emp
where salary>3000 and
bonus is not null;

D037.查询月工资大于3500但是小于5000的员工信息

select* from emp
where salary>3000
and salary<5000;

D038.查询编号是1001、1003、1004的员工信息

select* from emp
where eid in (1001 , 1003 , 1004);

D039.分组统计

在my_student表中，根据性别分组，使用函数统计出男女分别的身高最高、最低、平均年龄、总年龄

select max(height),min(height),avg(grade),sum(grade) 
from student group by gender;

D040.在my_student表中，按照性别分组，查询出性别、个数、该性别的所有姓名。如（male| 2| 张三，王五）

select gender,count(*),name from student group by gender;
