110个Database练习任务(一)

发表于 2016-10-02 | 分类于 Database | | 阅读次数:
字数统计: 554 | 阅读时长 ≈ 3

本文通过110个Database的练习任务及详细的任务解答帮助你走进数据库的大门！！！

D001．查看所有数据库

show databases;

D002.新增数据库

数据库名是”mydatabase”，字符集是”utf8”

create database mydatabase char set utf8;

D003.删除数据库

删除数据库”mydatabase”

drop database mydatabase;

D004.进入数据库环境

使用”mydatabase” 

create database mydatabase;
use mydatabase;

D005.新增数据库表

创建数据库表student

表字段有 name varchar(10),

gender varchar(10),num varchar(18) ,age int(10),字符集为utf8(首先要使用mydatabase,否则会报错)。

use mydatabase;
create table student(
name varchar(10),gender varchar(10),num varchar(18) ,age int(10)
) char set utf8;

D006.查看所有表

use mydatabase;
show tables;

D007.查看表结构

查看student表结构

use mydatabase;
describe student;

D008.修改表名

将student表改名为my_student

use mydatabase;
alter table student rename my_student;

D009.改my_student表增加id列在第一字段

use mydatabase;
alter table my_student add id varchar(10) first;
desc my_student;

D010.删除字段

删除my_student表age字段

use mydatabase;
alter table my_student drop age;
desc my_student;;

D012.新增数据

向my_student表中插入（001,male,Tom,3）

（002,female,Lily,4）

use mydatabase;
insert into my_student()
values("001","male","Tom","3");
insert into my_student()
values("002","female","Lily","4");

D013.查看my_student中的所有数据

use mydatabase;
select* from my_student;

D014.查看my_student中满足id为001的数据

use mydatabase;
select* from my_student where(id=001);

D015.删除数据

删除my_student表中gender为’male’的数据

use mydatabase;
delete from my_student
where(gender="male");
select* from my_student;

D016.修改my_student表，设置id为主键(用关键字change)

use mydatabase;
alter table my_student change id id int (10) primary key;

D017.将my_student表中主键删除

use mydatabase;
alter table my_student drop primary key;

D018.添加主键

更改my_student表中id主键为自增

use mydatabase;
alter table my_student add primary key(id);
alter table my_student modify id int auto_increment;

D019.将my_student表中name字段设置成唯一键

use mydatabase;
alter table my_student add unique(name);
desc my_student;

D011.删除数据表my_student

use mydatabase;
drop table my_student;
show tables;

D020.修改字段名

修改my_student表字段gender为sex

use mydatabase;
alter table my_student change  gender sex varchar(10);
desc my_student;

D021.更新数据

将my_student表中名为Tom的学生改为female

use mydatabase;
update my_student
set sex = "female"
where name = "Tom";
select* from my_student;

D022.将my_student表中id字段设置为主键(用关键字modify)

use mydatabase;
alter table my_student drop id;
alter table my_student add id int not null primary key Auto_increment; 
alter table my_student modify id int auto_increment;
