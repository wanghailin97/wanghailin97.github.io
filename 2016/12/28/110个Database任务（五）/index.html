database:company

create database company;
use company;
##部门表dept： 部门编号(deptno)，部门名称(dname)，地址(loc)； 
CREATE TABLE dept( 

deptno INT PRIMARY KEY,dname VARCHAR(20),loc VARCHAR(20) 

);

##员工表emp: 员工编号(empno)，员工名字(ename)，职务(job)，管理者编号(mgp)，
##入职日期(hiredate)，月薪资(sal)，月奖金(comm)，部门编号(deptno)； 
CREATE TABLE emp(empno INT PRIMARY KEY,ename VARCHAR(20) NOT NULL, 

job VARCHAR(20) CHECK (job IN('CLERK','SALESMAN','MANAGER','ANALYST')),mgp INT, 

hiredate DATETIME ,sal DECIMAL(10,2),comm DECIMAL(10,2), 

deptno INT, 

CONSTRAINT pk_we FOREIGN KEY (deptno) REFERENCES dept (deptno) 

);

D086.取得每个部门部门编号和最高薪水

use company;
select d.deptno,max(ifnull(e.sal,0)+ifnull(e.comm,0)) from dept d
left join emp e on d.deptno=e.deptno
group by d.deptno;

D087.不准用组函数（max),取得最高薪水

use company;

##法一
select ifnull(e.sal,0)  as "最高薪水" from emp e
order by ifnull(e.sal,0) desc limit 1;

##法二
select sal from emp 
where sal not in(
select distinct e1.sal from emp e1
join emp e2 on e1.sal < e2.sal
);

D088.取得薪水最高的前五名员工

use company;
select e.ename as "姓名",ifnull(e.sal,0)  as "薪水" from emp e
order by ifnull(e.sal,0) desc limit 5;

D089.取得薪水最高的第六到第十名员工

use company;
select e.* from emp e
order by e.sal desc limit 5,5;

D090.取得最后入职的五名员工

use company;
select e.* from emp e 
order by e.hiredate desc limit 5;

D091.取得每个薪水等级有多少员工

use company;

#step 1
select e.ename,s.grade from emp e
left join salgrade s on e.sal between s.losal and s.misal
order by s.grade;

#step 2
select t.grade,count(t.grade) as "数量" from
(select e.ename,s.grade from emp e
left join salgrade s on e.sal between s.losal and s.misal
order by s.grade) as t
group by t.grade;

D092.列出所有员工及领导的名字

use company;
select e1.ename as "员工",e2.ename as "领导" from emp e1,emp e2
where e1.mgp=e2.empno group by e1.empno;

D093.列出受雇日期早于其直接上级的所有员工编号、姓名、部门名称

use company;
select e.empno,e.ename,d.dname from emp e,emp b,dept d 
where e.mgp=b.empno
and d.`deptno`=e.`deptno` 
and e.hiredate>b.`hiredate`;

D094.列出部门名称和这些部门的员工信息，并列出没有员工的部门名称

use company;
select d.dname,e.empno,e.ename from dept d 
left join emp e on d.deptno = e.deptno;

D095.列出至少有五个员工的部门

use company;
select e.deptno ,count(1) totalEmp from emp e 
group by e.deptno having totalEmp >=5;

D096.列出与最低薪水大于1500的各种工作名称及从事此工作的最低工资和员工人数

use company;
select e.job,min(e.sal) minsal ,count(1) totalEmp from emp e 
group by e.job having minsal>1500;

D097.列出在部门“SALES”<销售部>工作的员工姓名

use company;
select ename from emp 
where deptno=(
select deptno from dept where dname='SALES');

D098.列出所有员工的年工资，按年薪从低到高排序

use company;
select ename,(sal+ifnull(comm,0))*12 yearsal from emp 
order by yearsal ASC;

D099.创建一个emp员工表的视图,视图名称为myview

use company;
create view myview as select* from emp;

D100.查询视图myview

use company;
select* from myview;

D101.删除视图myview

use  company;
delete from myview;

D102.创建表dept的视图deptview ， 并为字段deptno,dname,loc分别命名别名为a,b,c

use company;
create view deptview 
as select deptno as a,dname as b,loc as c from dept;

D103.修改视图deptview,并将字段a,b,c修改为dno,dname,dloc

use company;
create or replace view deptview 
as select deptno as dno,dname as dname,loc as dloc from dept;

D104.哪些人的薪水在部门平均薪水之上，显示姓名，工资，部门编号，部门平均工资

use company;
select e2.ename,e2.sal ,e2.`deptno`,a.avgsal from emp e2 
join (
select e1.deptno ,avg(sal) avgsal from emp e1 group by deptno
) a on e2.deptno =a.deptno 
and e2.sal>a.avgsal;

D105.取得部门的编号、平均薪水、所处等级

use company;
select a.deptno,a.avgsal,s.grade from salgrade s 
join (
select emp.deptno ,avg(sal) avgsal from emp 
group by deptno) a on a.avgsal>s.losal 
and a.avgsal;

D106.取得平均薪水最高的部门的部门编号

use company;
select deptno from emp 
group by deptno having avg(sal) =(
select avg(sal) from emp 
group by deptno desc limit 1);

D107.取得比普通员工的最高薪资还要高的经理人名称

use company;
select ename from emp 
where sal> ( 
select max(sal) maxsal from emp 
where empno not in (
select distinct mgp from emp 
where mgp is not null));

D108.列出薪金高于公司平均薪水的所有员工名，所在部门名、上级领导名、员工的工资等级

use company;
select d.dname, e.ename, b.ename as leadername, s.grade from emp e 
join dept d on e.deptno=d.deptno 
left join emp b on e.mgp =b.`empno` 
join salgrade s on e.sal between s.losal and s.misal 
where e.sal>(select avg(sal) avgsal from emp);

D109.列出薪金等于部门30中员工的薪金的其他部门员工的姓名和薪金

use company;
select ename,sal from emp 
where sal in (
select sal from emp 
where deptno=30) 
and deptno<>30;

D110.列出薪金高于在部门30工作的所有员工的薪金的员工姓名和薪金、部门名称

use company;
select ename,sal,dname from emp 
join dept on emp.`deptno`=dept.`deptno` 
where emp.sal>(
select max(sal) maxsal from emp 
where deptno=30 );
