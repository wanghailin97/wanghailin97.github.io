D061.查询至少有一门课与学号为”01”的同学所学相同课程的同学的信息

use school;
#法一
select distinct s.* from student s
left join sc on sc.sid=s.sid
where s.sid!=01
and sc.cid in (01,02,03);

#法二
select distinct s.* from student s,sc
where s.sid=sc.sid 
and sc.cid in (
select cid from sc where sc.sid=01)
and s.sid!=01;

D062.查询两门及其以上不及格课程的同学的学号，姓名及其平均成绩

use school;
select s.sid,s.sname,avg(sc.score) as "平均成绩" from student s,sc
where sc.sid=s.sid
and s.sid in (
select sc.sid from sc 
where sc.score<60 group by sc.sid having count(sc.cid)>=2)
group by s.sid,sc.sid;

select s.sid,s.sname,avg(sc.score) as "平均成绩" from student s,sc
where sc.sid=s.sid
and s.sid in (
select sc.sid from sc 
where sc.score<60 group by sc.sid having count(1)>=2)
group by s.sid,sc.sid;

D063.检索”01”课程分数小于60，按分数降序排列的学生信息

use school;
select s.*,sc.score as "01课程分数" from student s,sc
where s.sid=sc.sid
and sc.score<60 
and sc.cid=01 group by sc.sid
order by sc.score desc;

D064.查询学生的总成绩和信息并进行排名

use school;
select s.*,sum(sc.score) as "总成绩" from student s,sc
where s.sid=sc.sid group by s.sid order by sum(sc.score) desc;

D065.查询老师所教课程的平均分、老师编号、老师姓名（按平均分从高到低显示）

use school;
select t.tid,t.tname,avg(sc.score) as "平均分数" from teacher t,sc,course c
where t.tid=c.tid
and c.cid = sc.cid group by sc.cid
order by avg(sc.score) desc;

D066.查询1990年出生的学生名单

use school;
select s.* from student s
where s.sbirth like '1990%'

D067.查询每门课程的编号，名字，老师编号，平均成绩，结果按平均成绩降序排列，平均成绩相同时，按课程编号升序

use school;
select c.*,avg(sc.score) as "平均成绩" from course c,sc
where c.cid=sc.cid group by c.cid
order by avg(sc.score) desc,c.cid;

D068.查询平均成绩大于等于85的所有学生的学号、姓名和平均成绩

use school;
select s.sid,s.sname,avg(sc.score) as "平均成绩" from student s,sc
where s.sid=sc.sid
group by s.sid,sc.sid
having avg(sc.score)>=85
order by avg(sc.score) desc;

D069. 查询课程名称为”数学”，且分数低于60的学生姓名和分数

use school;
select s.sname,sc.score from student s,sc
where s.sid=sc.sid
and sc.cid in (
select c.cid from course c where c.cname="数学")
order by sc.score desc;

#法一
select s.*,c.cid,c.cname,sc.score from sc 
inner join student s on s.sid=sc.sid 
inner join course c on sc.cid=c.cid;

#法二
select s.*,sc1.score as "语文成绩",sc2.score as "数学成绩",sc3.score as "英语成绩"
from student s
left join sc sc1 on sc1.sid=s.sid and sc1.cid=01
left join sc sc2 on sc2.sid=s.sid and sc2.cid=02
left join sc sc3 on sc3.sid=s.sid and sc3.cid=03;

D071.查询任何一门课程成绩在70分以上的学生姓名、课程名称和分数

use school;
select s.sname,c.cname,sc.score from sc
inner join student s on s.sid=sc.sid
inner join course c on sc.cid=c.cid
where sc.score>70
and count(c.cname)=3 group by c.cname;

select s.sname,c.cname,sc.score from student s,sc,course c
where s.sid=sc.sid 
and sc.cid=c.cid
and sc.score>=70;

D072.查询课程编号为01且课程成绩在80分及以上的学生的学号和姓名

use school;
select s.sid,s.sname,sc.score from student s
left join sc on sc.sid=s.sid
where sc.cid=01 and sc.score>=80;

D073.查询每门课程的编号和学生人数

use school;
select sc.cid,count(sc.sid) as "学生人数" from sc
group by sc.cid;

D074.查询各学生的信息以及他们的年龄

use school;
select s.*,year(now())-s.sbirth as "年龄" from student s;

D075.为sc表加外键 sid 被student表的sid字段约束

use school;
desc sc;
alter table sc engine = InnoDB;
alter table `sc` add constraint `fk_sid` foreign key(`sid`) references `student`(`sid`);

D076.删除sc表的sid外键

use school;
alter table sc drop foreign key sid;

D077.查询没学过”张三”老师讲授的任一门课程的学生信息

use school;
select s.* from student s
where s.sid not in (
select sc.sid from sc 
left join course c on c.cid=sc.cid
left join teacher t on t.tid=c.tid
where t.tname="张三"
) group by s.sid;

D078.按平均成绩从高到低显示所有学生的信息及所学课程的成绩以及平均成绩

use school;
/*select s.*,sc1.score as "语文", sc2.score as "数学", 
sc3.score as "英语", avg(sc4.score) as "平均成绩"
from student s,sc sc1,sc sc2,sc sc3,sc sc4
where s.sid=sc1.sid and s.sid=sc2.sid and s.sid=sc3.sid
and sc1.cid=01 and sc2.cid=02 and sc3.cid=03
group by s.sid;*/
##这个只能检索三科都学过的学生成绩

select s.*,sc1.score as "语文", sc2.score as "数学", 
sc3.score as "英语", avg(sc4.score) as "平均成绩"
from student s
left join sc sc1 on sc1.sid=s.sid
and sc1.cid=01 
left join sc sc2 on sc2.sid=s.sid
and sc2.cid=02  
left join sc sc3 on sc3.sid=s.sid
and sc3.cid=03 
left join sc sc4 on sc4.sid=s.sid
group by s.sid;

D079.查询学生的信息、平均成绩及其名次

use school;
select t.*, px = rank() over(order by avg(sc.score) desc) from
(
select s.*,avg(sc.score) as "平均成绩"
from student s left join sc on sc.sid=s.sid
group by s.sid
) as t
order by px;

select s.*,avg(sc.score) as "平均成绩"
from student s 
left join sc on sc.sid=s.sid
group by s.sid;

D080.查询各科成绩前三名的学生编号，课程编号，成绩，名次

use school;
select* from (
select t.* , px = (select count(distinct score) from sc where cid = t.cid and score >= t.score) from sc t) m 
where px between 1 and 3 order by m.cid , m.px;

D081.查询出只有两门课程的全部学生的信息和所学课程数

use school;
select s.*, count(sc.cid) as "课程数量" from student s
left join sc on s.sid=sc.sid
group by sc.sid having count(sc.cid)=2;

D082.查询选修”张三”老师所授课程的学生中，成绩最高的学生信息及其成绩

use school;
select s.*, sc.score from student s
left join sc on s.sid=sc.sid
left join course c on sc.cid=c.cid
left join teacher t on c.tid=t.tid
where t.tname="张三" 
order by sc.score desc limit 1;

D083.查询不同课程成绩相同的学生的学生编号、课程编号、学生成绩

use school;
select sc.* from sc,
(select cid, score from sc group by cid,score having count(1)>1) as n
where sc.cid=n.cid and sc.score=n.score order by sc.cid,sc.score,sc.sid;

D084.查询每门课程成绩最好的前两名 的学号，课程编号，成绩

use school;
select sc1.* from sc sc1 
where score in (
select score from sc sc2 
where sc1.cid = sc2.cid 
order by score desc limit 2
) 
order by sc1.cid,sc1.score desc;

D085.统计每门课程的学生选修人数（超过5人的课程才统计）。要求输出课程号和选修人数，查询结果按人数降序排列，若人数相同，按课程号升序排列

use school;
select course.cid , course.cname , count(*) as "学生人数"
from course , sc
where course.cid = sc.cid
group by  course.cid , course.cname
having count(*) >= 5
order by count(*) desc , course.cid;
