D041.在emp表中，查询编号不是1001、1003、1004的员工信息

use mydatabase;
select* from emp
where eid not in (1001,1003,1005);

D042.在emp表中，查询员工姓名是3个字的员工信息

use mydatabase;
select* from emp
where char_length(ename)=3;

D043.在emp表中,查询姓张的员工信息

use mydatabase;
select* from emp
where ename regexp '^张';

D044.在emp表中,查询出在99年雇佣的员工信息

use mydatabase;
select* from emp
where hiretime regexp '^1999';

D045.在emp表中,查询出员工工资没有包含6和8的员工信息

use mydatabase;
select* from emp
where salary  not like '%6%' 
and salary not like '%8%';

D046.在emp表中,按照工资由高到低查询员工信息

use mydatabase;
select* from emp
order by salary desc;

D047.在emp表中,要求查询出101部门的所有雇员信息，查询的信息按照工资由高到低排序，如果工资相等，则按照雇佣日期由早到晚排序

use mydatabase;
select* from emp
where depart=101 order by salary desc,hiretime asc;

D048.查询101部门的所有员工信息，并显示所在部门名称

use mydatabase;
select emp.eid,ename,title,hiretime,salary,bonus,depart,dept.dname as "部门名称"
from emp,dept
where emp.depart=dept.did and emp.depart=101;

D049. 查询1001员工的部门领导信息

use mydatabase;
select* from emp 
where eid = (
select dieader from dept 
where did = (
select depart from emp 
where eid = 1001
));

D050. 查询部门员工数量，平均工资，最低工资及最低工资的员工姓名

use mydatabase;
select depart as "部门名称",count(*) as "员工数量",
avg(salary) as "平均工资",min(salary) as "最低工资",
ename as "最低工资员工姓名" from emp
group by depart;

D051.查询”01”课程比”02”课程成绩低的学生的信息及课程01和02的分数

use school;
select a.* , b.score as "课程'01'的分数",c.score as "课程'02'的分数" 
from student a
left join sc b on a.sid = b.sid and b.cid = '01'
left join sc c on a.sid = c.sid and c.cid = '02'
where ifnull(b.score,0) < c.score;

D052.查询平均成绩大于等于60分的同学的学生编号和学生姓名和平均成绩

use school;
select student.sid as "编号",student.sname as "姓名",
avg(sc.score) as "平均成绩" from student,sc
where sc.sid=student.sid
group by student.sid,sc.sid
having avg(sc.score)>=60;

D053.查询所有同学的学生编号、学生姓名、选课总数、所有课程的总成绩

use school;
select student.sid,sname,count(sc.cid) as "选课总数",
sum(sc.score) as "总成绩" from student,sc
where student.sid=sc.sid
group by student.sid,sc.sid;

D054.查询”李”姓老师的数量

use school;
select count(t.tname) as "李姓老师数量" from teacher t
where tname like '李%';

D055.查询学过”张三”老师授课的同学的信息

use school;
select s.* from student s, teacher t, course c, sc
where s.sid = sc.sid and sc.cid = 02
group by s.sid;

select s.* from student s, teacher t, course c, sc
where s.sid = sc.sid and sc.cid = (
select c.cid where c.tid=01)
group by s.sid;

select s.* from student s, teacher t, course c, sc
where s.sid = sc.sid and sc.cid = (
select c.cid where c.tid=(
select t.tid where t.tname="张三"
))
group by s.sid;

D056.查询没学过”张三”老师授课的同学的信息

use school;
#法一
select s.* from student s
where s.sid not in(
select sc.sid from sc where sc.cid=02)
;
#法二
select s.* from student s
where s.sid not in(
select sc.sid from sc where sc.cid=(
select c.cid from course c where c.tid=01
));
#法三
select s.* from student s
where s.sid not in(
select sc.sid from sc where sc.cid=(
select c.cid from course c where c.tid=(
select t.tid from teacher t where t.tname="张三"
)));

D057. 查询学过编号为”01”并且也学过编号为”02”的课程的同学的信息

use school;
#法一
select s.* from student s, sc 
where s.sid =sc.sid and sc.cid = 01 
and exists (select 1 from sc sc_2 where sc_2.sid = sc.sid 
and sc_2.cid = 02);

#法二
select s.* from student s, sc 
where s.sid =sc.sid and sc.cid = 02 
and exists (select 2 from sc sc_2 where sc_2.sid = sc.sid 
and sc_2.cid = 01);

D058. 查询没有学全所有课程的同学的信息

use school;
select s.* from student s,sc
where s.sid = sc.sid
group by sc.sid
having count(sc.cid)< 3;

D059.查询”01”课程比”02”课程成绩高的学生的信息及课程分数

use school; 
select a.* , b.score as "课程'01'的分数",c.score as "课程'02'的分数" 
from student a
left join sc b on a.sid = b.sid and b.cid = '01'
left join sc c on a.sid = c.sid and c.cid = '02'
where b.score > ifnull(c.score,0);

D060.查询学过编号为”01”但是没有学过编号为”02”的课程的同学

use school;
select s.* from student s, sc 
where s.sid =sc.sid and sc.cid = 01 
and not exists (select 01 from sc sc_2 where sc_2.sid = sc.sid 
and sc_2.cid = 02);
